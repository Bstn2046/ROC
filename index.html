Core01_ARN ‚Äî volledige configuratie

Dit blok zet poorten naar switchport-mode, maakt VLAN‚Äôs & SVI‚Äôs, DHCP, trunks, ACL (guest isolatie), OSPF, en STP-root. Verander dit in kladblok

Cisco IOS ‚Äî Core01_ARN

enable
configure terminal
hostname Core01_ARN

! Alle poorten naar L2-switchport (veilig op L3-switch)
interface range fastethernet0/1 - 24
 switchport
exit
interface range gigabitethernet0/1 - 2
 switchport
exit

! VTP Server
vtp domain SOLAR
vtp mode server

! VLANs
vlan 30
 name Productie
vlan 40
 name Marketing
vlan 50
 name Gasten_Wifi
exit

! SVI's
interface vlan 1
 ip address 10.10.0.2 255.255.255.0
 no shutdown
interface vlan 30
 ip address 10.10.30.1 255.255.255.0
 no shutdown
interface vlan 40
 ip address 10.10.40.1 255.255.255.0
 no shutdown
interface vlan 50
 ip address 10.10.50.1 255.255.255.0
 no shutdown

! DHCP pools
ip dhcp excluded-address 10.10.30.1 10.10.30.50
ip dhcp excluded-address 10.10.40.1 10.10.40.50
ip dhcp excluded-address 10.10.50.1 10.10.50.50

ip dhcp pool vlan30
 network 10.10.30.0 255.255.255.0
 default-router 10.10.30.1
 dns-server 62.36.55.85

ip dhcp pool vlan40
 network 10.10.40.0 255.255.255.0
 default-router 10.10.40.1
 dns-server 62.36.55.85

ip dhcp pool vlan50
 network 10.10.50.0 255.255.255.0
 default-router 10.10.50.1
 dns-server 62.36.55.85

! Trunks
interface fastethernet0/1
 description Trunk-to-SW01-ARN
 switchport mode trunk
 switchport trunk allowed vlan 1,30,40,50

interface fastethernet0/3
 description Trunk-to-SW02-ARN
 switchport mode trunk
 switchport trunk allowed vlan 1,30,40,50

! Access-poort voor AP_Gast (VLAN50)
interface fastethernet0/24
 description Access-to-AP_Gast
 switchport mode access
 switchport access vlan 50

! OSPF
router ospf 1
 network 10.0.0.0 0.255.255.255 area 0

! ACL (guest isolatie) op VLAN50 inbound
access-list 150 deny ip 10.10.50.0 0.0.0.255 10.10.30.0 0.0.0.255
access-list 150 deny ip 10.10.50.0 0.0.0.255 10.10.40.0 0.0.0.255
access-list 150 deny ip 10.10.50.0 0.0.0.255 172.16.10.0 0.0.0.255
access-list 150 deny ip 10.10.50.0 0.0.0.255 172.16.5.0 0.0.0.255
access-list 150 deny ip 10.10.50.0 0.0.0.255 172.16.20.0 0.0.0.255
access-list 150 permit ip any any

interface vlan 50
 ip access-group 150 in

! STP Root voor alle VLANs
spanning-tree vlan 1,30,40,50 priority 4096

end
Tip: Controleer met show vlan brief en show interfaces trunk. Fa0/24 moet bij VLAN 50 staan, Fa0/1 en Fa0/3 moeten trunk zijn.
